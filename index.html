JS helper functions

// Easy looping:
function each(arr, fn, scope, args) {
	for (var i = 0, l = arr.length; i < l; i++) {
		if (args) fn.apply([arr[i]].concat(args));
		else fn.call(scope, arr[i], i, arr);
	}
}

// Get elements by CSS selector:
function qs(selector, scope) {
	return (scope || document).querySelector(selector);
}
function qsa(selector, scope) {
	return (scope || document).querySelectorAll(selector);
}

// addEventListener wrapper:
function on(target, type, callback) {
	if (target.length) each(target, function(target) {
		on(target, type, callback);
	});
	else target.addEventListener(type, callback, false);
}

"Throttle" function calls with a debounce function

function debounce(fn, delay, timer) {
	delay = delay || 150;
	return function() {
		var args = arguments,
			context = this;
		clearTimeout(timer);
		timer = setTimeout(function() {
			fn.apply(context, args);
		}, delay);
	};
}

$$('.tabbed-panels').forEach(function(parent) {
	var tabs = $$('.tab', parent);
	var panels = $$('.panel', parent);

Modular width classes (for those who hate media queries)

(function(window) { // modular width classes:
	'use strict';

	var document = window.document;
	var gtEls = [].slice.call(document.querySelectorAll('[data-gt]'));
	var ltEls = [].slice.call(document.querySelectorAll('[data-lt]'));
	var gt = gtEls.map(function(el) {
		return el.getAttribute('data-gt').split(' ').map(Number);
	});
	var lt = ltEls.map(function(el) {
		return el.getAttribute('data-lt').split(' ').map(Number);
	});

	function resizer() {
		gtEls.forEach(function(el, i) {
			var w = el.offsetWidth;
			gt[i].forEach(function(bp) {
				el.classList[(w > bp) ? 'add' : 'remove']('gt' + bp);
			});
		});
		ltEls.forEach(function(el, i) {
			var w = el.offsetWidth;
			lt[i].forEach(function(bp) {
				el.classList[(w < bp) ? 'add' : 'remove']('lt' + bp);
			});
		});
	}

	resizer();
	window[window.addEventListener ? 'addEventListener' : 'attachEvent']('resize', resizer, false);
})(window);

function getNamedArgsObj() {
	var fn = arguments.callee.caller;
	var s = fn.toString().split('(')[1], obj = {};
	s.split(')')[0].split(',').forEach(function(param, i) {

Maxified:

// feature detection:
(function(h,f) {
	h.className = 'js';
	ok = function(p,e,c) {
		var c = c || p;
		e && !f[c] && (h.className += ' ' + ((f[c] = p in document.createElement(e)) ? c : 'no-' + c));
		return f[c];
	};
})(document.documentElement, {});

How To

Use it like this:

if( !ok('placeholder','textarea','textareaPlaceholder') ) {
	// textarea placeholder polyfill

	if( !ok('placeholder','input','inputPlaceholder') ) {
		// input placeholder polyfill
	}
}

After each ok() function is evaluated, a class will be added to the <html> element with the value of the ok() function's third argument (prefixed with 'no-' if not supported). You can then add feature specific CSS styles:

.inputPlaceholder input[placeholder] + label {
	display: none;
}

Select text with Javascript

codeElements = document.getElementsByTagName('code');

for (var i = codeElements.length; i--;) {
	codeElements[i].onclick = function() {
		if (document.selection) {
			var range = document.body.createTextRange();
			range.moveToElementText(this);
			range.select();
		}
		else if (window.getSelection) {
			var range = document.createRange();
			range.selectNode(this);
			window.getSelection().addRange(range);
		}
	}
}

Notes:

This will select the text in the <code> element that the user clicks on.

For textareas and input elements use this instead:

element.onclick = function() {
    this.select();
}
